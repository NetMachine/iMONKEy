function ajustarImagen() { const contenedorImagen = document.getElementById('contenedor'); const imagen = contenedorImagen.querySelector('img'); if (!contenedorImagen || !imagen) { console.error('No se encontraron los elementos necesarios para ajustar la imagen'); return; } const anchoContenedor = contenedorImagen.clientWidth; const altoContenedor = contenedorImagen.clientHeight; const relacionAspectoImagen = imagen.naturalWidth / imagen.naturalHeight; let nuevoAncho = Math.floor(anchoContenedor / relacionAspectoImagen); let nuevoAlto = Math.floor(anchoContenedor / relacionAspectoImagen * relacionAspectoImagen); if (nuevoAlto > altoContenedor) { nuevoAlto = altoContenedor; nuevoAncho = Math.floor(nuevoAlto * relacionAspectoImagen); } imagen.style.width = nuevoAncho + 'px'; imagen.style.height = nuevoAlto + 'px'; imagen.style.left = Math.floor((anchoContenedor - nuevoAncho) / 2) + 'px'; imagen.style.top = Math.floor((altoContenedor - nuevoAlto) / 2) + 'px'; } window.addEventListener('resize', ajustarImagen); ajustarImagen(); // Ejecutar la función al cargar la página Imagen centrada responsive
